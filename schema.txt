-- Financial transfers & expenses — schema, indexes, RLS
-- Run in Supabase SQL Editor or via supabase db push

-- Extensions
create extension if not exists "pgcrypto";

-- Transactions (حوالات)
create table if not exists public.transactions (
  id uuid primary key default gen_random_uuid(),
  user_id uuid not null references auth.users (id) on delete cascade,
  sender text not null,
  beneficiary text not null,
  amount numeric(14, 2) not null check (amount > 0),
  transaction_date date not null,
  category text default 'general',
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);

create index if not exists idx_transactions_user_id on public.transactions (user_id);
create index if not exists idx_transactions_transaction_date on public.transactions (transaction_date desc);
create index if not exists idx_transactions_user_date on public.transactions (user_id, transaction_date desc);
create index if not exists idx_transactions_created_at on public.transactions (created_at desc);

-- Expenses (مصروفات)
create table if not exists public.expenses (
  id uuid primary key default gen_random_uuid(),
  user_id uuid not null references auth.users (id) on delete cascade,
  description text not null,
  amount numeric(14, 2) not null check (amount > 0),
  expense_date date not null,
  category text default 'general',
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);

create index if not exists idx_expenses_user_id on public.expenses (user_id);
create index if not exists idx_expenses_expense_date on public.expenses (expense_date desc);
create index if not exists idx_expenses_user_date on public.expenses (user_id, expense_date desc);
create index if not exists idx_expenses_created_at on public.expenses (created_at desc);

-- updated_at trigger helper
create or replace function public.set_updated_at()
returns trigger
language plpgsql
as $$
begin
  new.updated_at = now();
  return new;
end;
$$;

create trigger tr_transactions_updated_at
  before update on public.transactions
  for each row execute function public.set_updated_at();

drop trigger if exists tr_expenses_updated_at on public.expenses;
create trigger tr_expenses_updated_at
  before update on public.expenses
  for each row execute function public.set_updated_at();

-- أدوار المستخدمين: user | manager | admin

-- جدول الملفات الشخصية
create table if not exists public.profiles (
  id uuid primary key references auth.users (id) on delete cascade,
  role text not null default 'user' check (role in ('user', 'manager', 'admin')),
  full_name text default '',
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);

create index if not exists idx_profiles_role on public.profiles (role);

drop trigger if exists tr_profiles_updated_at on public.profiles;
create trigger tr_profiles_updated_at
  before update on public.profiles
  for each row execute function public.set_updated_at();

alter table public.profiles enable row level security;

-- دوال للتحقق من الدور (SECURITY DEFINER لتفادي تعارض RLS مع جدول profiles)
create or replace function public.is_staff()
returns boolean
language plpgsql
stable
security definer
set search_path = public
as $$
begin
  return exists (
    select 1 from public.profiles p
    where p.id = auth.uid()
      and p.role in ('manager', 'admin')
  );
end;
$$;

create or replace function public.is_admin()
returns boolean
language plpgsql
stable
security definer
set search_path = public
as $$
begin
  return exists (
    select 1 from public.profiles p
    where p.id = auth.uid() and p.role = 'admin'
  );
end;
$$;

grant execute on function public.is_staff() to authenticated;
grant execute on function public.is_admin() to authenticated;

  on public.profiles for update

    and role = (select p2.role from public.profiles p2 where p2.id = auth.uid())
  );

  on public.profiles for update
  using (public.is_admin())
  with check (true);

-- إنشاء profile تلقائياً عند تسجيل مستخدم في Auth
create or replace function public.handle_new_user()
returns trigger
language plpgsql
security definer
set search_path = public
as $$
begin
  insert into public.profiles (id, role, full_name)
  values (
    new.id,
    'user',
    coalesce(new.raw_user_meta_data->>'full_name', '')
  )
  on conflict (id) do nothing;
  return new;
end;
$$;

drop trigger if exists on_auth_user_created on auth.users;
create trigger on_auth_user_created
  after insert on auth.users
  for each row execute function public.handle_new_user();

-- مستخدمون قدامى بلا صف profile
insert into public.profiles (id, role, full_name)
select u.id, 'user', coalesce(u.raw_user_meta_data->>'full_name', '')
from auth.users u
where not exists (select 1 from public.profiles p where p.id = u.id)
on conflict (id) do nothing;

-- ربط المصروفات بالحوالات (إجبارية) + العملة على الحوالة ومزامنتها للمصروف
-- العملات: SAR (ر.س) · YER (ر.ي) · AED (د)

ALTER TABLE public.transactions
  ADD COLUMN IF NOT EXISTS currency text NOT NULL DEFAULT 'SAR';

ALTER TABLE public.transactions DROP CONSTRAINT IF EXISTS transactions_currency_check;
ALTER TABLE public.transactions ADD CONSTRAINT transactions_currency_check
  CHECK (currency IN ('SAR', 'YER', 'AED'));

  ADD COLUMN IF NOT EXISTS transaction_id uuid REFERENCES public.transactions (id) ON DELETE CASCADE;

  ADD COLUMN IF NOT EXISTS currency text NOT NULL DEFAULT 'SAR';

  CHECK (currency IN ('SAR', 'YER', 'AED'));

-- ربط المصروفات القديمة بأقدم حوالة لنفس المستخدم (إن وُجدت)
UPDATE public.expenses e
SET transaction_id = sub.id
FROM (
  SELECT DISTINCT ON (user_id) id, user_id
  FROM public.transactions
  ORDER BY user_id, created_at ASC
) sub
WHERE e.user_id = sub.user_id
  AND e.transaction_id IS NULL;

-- حذف مصروفات لا يمكن ربطها (لا توجد حوالات للمستخدم)
DELETE FROM public.expenses
WHERE transaction_id IS NULL;

ALTER TABLE public.expenses ALTER COLUMN transaction_id SET NOT NULL;

UPDATE public.expenses e
SET currency = t.currency
FROM public.transactions t
WHERE e.transaction_id = t.id;

CREATE INDEX IF NOT EXISTS idx_expenses_transaction_id ON public.expenses (transaction_id);
CREATE INDEX IF NOT EXISTS idx_expenses_user_transaction ON public.expenses (user_id, transaction_id);

-- قبل الإدراج/تغيير الحوالة: تعيين عملة المصروف من الحوالة
CREATE OR REPLACE FUNCTION public.sync_expense_currency_from_transaction()
RETURNS trigger
LANGUAGE plpgsql
AS $$
BEGIN
  SELECT t.currency INTO NEW.currency
  FROM public.transactions t
  WHERE t.id = NEW.transaction_id;
  IF NEW.currency IS NULL THEN
    RAISE EXCEPTION 'transaction not found for expense';
  END IF;
  RETURN NEW;
END;
$$;

DROP TRIGGER IF EXISTS tr_expenses_sync_currency ON public.expenses;
CREATE TRIGGER tr_expenses_sync_currency
  BEFORE INSERT OR UPDATE OF transaction_id ON public.expenses
  for each row execute function public.sync_expense_currency_from_transaction();

-- عند تغيير عملة الحوالة: تحديث مصروفاتها
CREATE OR REPLACE FUNCTION public.cascade_transaction_currency_to_expenses()
RETURNS trigger
LANGUAGE plpgsql
AS $$
BEGIN
  IF NEW.currency IS DISTINCT FROM OLD.currency THEN
    UPDATE public.expenses
    SET currency = NEW.currency
    WHERE transaction_id = NEW.id;
  END IF;
  RETURN NEW;
END;
$$;

DROP TRIGGER IF EXISTS tr_transactions_currency_cascade ON public.transactions;
CREATE TRIGGER tr_transactions_currency_cascade
  AFTER UPDATE OF currency ON public.transactions
  for each row execute function public.cascade_transaction_currency_to_expenses();

COMMENT ON COLUMN public.transactions.currency IS 'SAR ر.س · YER ر.ي · AED د';

COMMENT ON COLUMN public.expenses.currency IS 'مزامن مع عملة الحوالة (للتقارير والعرض)';
 







-- مهام داخلية: إنشاء/تعيين من قبل manager أو admin، تتبع عبر task_events، إشعارات بريد من Edge.

create table if not exists public.tasks (
  id uuid primary key default gen_random_uuid(),
  title text not null,
  description text not null default '',
  priority text not null default 'medium' check (priority in ('low', 'medium', 'high', 'urgent')),
  status text not null default 'pending' check (status in ('pending', 'in_progress', 'blocked', 'completed', 'cancelled')),
  due_at timestamptz,
  created_by uuid not null references auth.users (id) on delete cascade,
  assigned_to uuid not null references auth.users (id) on delete cascade,
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);

create index if not exists idx_tasks_assigned_to on public.tasks (assigned_to);
create index if not exists idx_tasks_created_by on public.tasks (created_by);
create index if not exists idx_tasks_status on public.tasks (status);
create index if not exists idx_tasks_due_at on public.tasks (due_at);
create index if not exists idx_tasks_updated_at on public.tasks (updated_at desc);

drop trigger if exists tr_tasks_updated_at on public.tasks;
create trigger tr_tasks_updated_at
  before update on public.tasks
  for each row execute function public.set_updated_at();

-- منع المكلف من تعديل حقول إدارية عبر واجهة مباشرة (التحقق النهائي في Edge أيضاً)
create or replace function public.enforce_task_assignee_update()
returns trigger
language plpgsql
security definer
set search_path = public
as $$
begin
  if public.is_staff() then
    return new;
  end if;
  if auth.uid() is distinct from old.assigned_to then
    raise exception 'غير مصرح' using errcode = '42501';
  end if;
  if new.title is distinct from old.title
     or new.priority is distinct from old.priority
     or new.due_at is distinct from old.due_at
     or new.assigned_to is distinct from old.assigned_to
     or new.created_by is distinct from old.created_by
  then
    raise exception 'يمكن للمكلف تعديل الحالة والوصف فقط' using errcode = '42501';
  end if;
  return new;
end;
$$;

drop trigger if exists tr_tasks_assignee_guard on public.tasks;
create trigger tr_tasks_assignee_guard
  before update on public.tasks
  for each row execute function public.enforce_task_assignee_update();

create index if not exists idx_task_events_task_id on public.task_events (task_id, created_at desc);

    actor_id = auth.uid()

          public.is_staff()
          or auth.uid() = t.assigned_to
